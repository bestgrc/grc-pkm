显示正好为五个字符的员工的姓名：
SELECT
	ENAME
FROM
	EMP
WHERE
	LENGTH (ENAME) = 5

显示不带R的员工姓名
SELECT
	ENAME
FROM
	EMP
WHERE
	ENAME  NOT IN (
		SELECT
			ENAME
		FROM
			EMP
		WHERE
			ENAME LIKE '%R%'
	)

显示所有员工姓名的前三个字符
SELECT
	SUBSTR(ENAME,1, 3)
FROM
	EMP

显示所有员工的姓名，用a替换所有A
SELECT
	"REPLACE"(ENAME , 'A','a')
FROM
	EMP

显示满10年雇佣年限的员工姓名和雇佣日期
SELECT
	ENAME , HIREDATE
FROM
	EMP
WHERE
	MONTHS_BETWEEN(SYSDATE,HIREDATE)/12 >37;

显示员工的详细资料，按姓名排序
SELECT
	*
FROM
	EMP
ORDER BY
	ENAME

显示所有员工的工作，姓名和薪金，按工作的降序排序，若工作相容则按薪金排序
SELECT
	"JOB",ENAME,SAL			//JOB是关键字所以要有引号
FROM
	EMP
ORDER BY
	"JOB" DESC,SAL

显示在一个月为30天的情况所有员工的日薪金，忽略余数。
SELECT
	ENAME,"TRUNC"(SAL/30) DAYSALE
FROM
	EMP

找出在任何年份的2月受雇的员工信息
SELECT
	*

	EMP

	TO_CHAR(HIREDATE,'mm')='02'

对于任何员工，显示其加入公司的天数

	EMP

查询出每个雇员的姓名，工资，部门名称，工资在公司的等级（salgrade），及其领导的姓名及工资所在公司的等级
SELECT
AE.ENAME,
 AE.SAL,
 DNAME,
 GRADE

FROM
	EMP AE,DEPT ,SALGRADE ,EMP BE
WHERE
	AE.DEPTNO=DEPT.DEPTNO AND
	AE.SAL>SALGRADE.LOSAL AND
	AE.SAL<SALGRADE.HISAL
ORDER BY
	GRADE

分组统计 group by           有group by，select后只能放groupby内容和组函数(不能出现分组函数之外的字段)
求出每个部门的雇员数量 按部门编号划分
SELECT	
	DEPTNO, count(*)
FROM 
	EMP
GROUP BY
	DEPTNO

每个部门平均工资
SELECT	
	DEPTNO,AVG(SAL)
FROM 
	EMP
GROUP BY
	DEPTNO

按部门分组，并显示部门的名称，及每个部门员工数
SELECT
	DNAME,"COUNT"(DNAME)
FROM
	DEPT,EMP
WHERE
	DEPT.DEPTNO=EMP.DEPTNO
GROUP BY 
	DEPT.DNAME
